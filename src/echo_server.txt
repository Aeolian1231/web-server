main():
while   接受新连接
    client_sock = accept(sock, (struct sockaddr *) &cli_addr, &cli_size);
    while   与客户端建立连接
        读入数据
        int readret = recv(client_sock, buf, BUF_SIZE, 0);
        数据处理并传回响应
        Send_back();

Send_back():
while   循环分解请求
    if  当前不是第一个请求
        查找\r\n\r\n位置
        nextRequest += 4; // 跳过\r\n\r\n
        while(跳过所有 \r 和 \n)
            if(当前位置是文件末尾)
                return
    解析http请求
    if  返回值为空
        返回400
    if  http版本不正确
        返回505
    if  请求类型未实现
        返回400
    if  请求是POST
        原样返回
    if  请求是GET
        if  uri合法(文件存在)
            打开文件并读取内容
            按照文件类型(html, css, png)构造响应头
            将响应头和文件内容合并返回
        if  文件不存在
            返回404
    if  请求是HEAD
        if  uri合法(文件存在)
            打开文件并读取内容
            按照文件类型(html, css, png)构造响应头
            返回响应头
        if  文件不存在
            返回404
